Side-by-Side Code Examples

Array Operations

NumPy

import numpy as np

# Create arrays
a = np.ones((3, 3), dtype=np.float32)
b = np.random.rand(3, 3).astype(np.float32)

# Operations
c = np.matmul(a, b)
d = c + 2.0
mean = np.mean(d)
sum_axis = np.sum(d, axis=0)

ndarray

open Ndarray

(* Create arrays *)
let a = Ndarray.ones [3; 3] float32
let b = Ndarray.random [3; 3] float32

(* Operations *)
let c = Ndarray.matmul a b
let d = Ndarray.add c (Ndarray.scalar 2.0 float32)
let mean = Ndarray.mean d
let sum_axis = Ndarray.sum d ~axis:[0]

Plotting

Matplotlib

import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(0, 10, 100)
y = np.sin(x)

plt.figure()
plt.plot(x, y, label='sin(x)')
plt.xlabel('x')
plt.ylabel('y')
plt.title('Sine Wave')
plt.legend()
plt.savefig('sine.png')

hugin

open Hugin
open Ndarray

let x = Ndarray.linspace 0.0 10.0 100 float64
let y = Ndarray.sin x

let fig = Figure.create ()
let ax = Figure.add_subplot fig 1 1 1
let _ = Axes.plot ax x y ~label:"sin(x)"
let _ = Axes.set_xlabel ax "x"
let _ = Axes.set_ylabel ax "y"
let _ = Axes.set_title ax "Sine Wave"
let _ = Axes.legend ax ()
let _ = Figure.savefig fig "sine.png"

Auto-differentiation

JAX

import jax
import jax.numpy as jnp

def f(x):
    return jnp.sum(x ** 2)

# Gradient
grad_f = jax.grad(f)
result = grad_f(jnp.array([1.0, 2.0, 3.0]))

# JIT compilation
jit_f = jax.jit(f)
compiled_result = jit_f(jnp.array([1.0, 2.0, 3.0]))

rune

open Rune

let f x = Tensor.sum (Tensor.pow x 2.0)

(* Gradient *)
let grad_f = grad f
let result = grad_f (Tensor.of_array [|1.0; 2.0; 3.0|])

(* JIT compilation *)
let jit_f = jit f
let compiled_result = jit_f (Tensor.of_array [|1.0; 2.0; 3.0|])

Key Differences

Raven Advantages

  • Type Safety: Compile-time guarantees prevent many runtime errors
  • Performance: Native compilation, no interpreter overhead
  • Memory Safety: No segfaults or memory leaks
  • Concurrency: OCaml 5 multicore and algebraic effects

Considerations

  • Learning Curve: OCaml syntax and functional programming concepts
  • Ecosystem: Smaller package ecosystem compared to Python
  • Pre-alpha: Some libraries still in development
  • Tooling: Different debugging and profiling tools
