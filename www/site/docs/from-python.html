Code Examples

Array Operations

NumPy

import numpy as np

# Create and manipulate arrays
a = np.ones((3, 3), dtype=np.float32)
b = np.random.rand(3, 3)
c = np.matmul(a, b) + 2.0

ndarray

open Ndarray

(* Create and manipulate arrays *)
let a = Ndarray.ones [3; 3] float32
let b = Ndarray.random [3; 3] float32
let c = Ndarray.(matmul a b + scalar 2.0 float32)

Auto-differentiation

JAX

import jax

def f(x):
return jnp.sum(x ** 2)

grad_f = jax.grad(f)
result = grad_f(jnp.array([1.0, 2.0]))

rune

open Rune

let f x = Tensor.sum (Tensor.square x)

let grad_f = grad f
let result = grad_f (Tensor.of_array [|1.0; 2.0|])

Key Differences

Raven Advantages

  • Type Safety: Compile-time error prevention
  • Performance: Native compilation, no interpreter
  • Memory Safety: No segfaults or memory leaks

Trade-offs

  • Learning Curve: OCaml syntax and concepts
  • Ecosystem: Smaller package ecosystem
  • Development: Some libraries still in beta
