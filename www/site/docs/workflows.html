Data Analysis Pipeline

Load CSV data, perform statistical analysis, and create publication-quality plots.

ndarray hugin ndarray-io 
(* Complete data analysis workflow *)
open Ndarray
open Hugin

(* 1. Load data from CSV *)
let data = Ndarray.Io.load_csv "sales_data.csv" ~has_header:true
let sales = Ndarray.get_column data 0
let months = Ndarray.get_column data 1

(* 2. Statistical analysis *)
let mean_sales = Ndarray.mean sales
let std_sales = Ndarray.std sales
let trend = Ndarray.polyfit months sales 1 (* linear trend *)

(* 3. Create visualizations *)
let fig = Figure.create ~figsize:(12.0, 8.0) ()

(* Main plot *)
let ax1 = Figure.add_subplot fig 2 2 1
let _ = Axes.plot ax1 months sales ~label:"Sales" ~linewidth:2.0
let _ = Axes.plot ax1 months (Ndarray.polyval trend months) ~label:"Trend" ~linestyle:"--"
let _ = Axes.set_title ax1 "Sales Over Time"
let _ = Axes.legend ax1 ()

(* Histogram *)
let ax2 = Figure.add_subplot fig 2 2 2
let _ = Axes.hist ax2 sales ~bins:20
let _ = Axes.set_title ax2 "Sales Distribution"

(* Box plot *)
let ax3 = Figure.add_subplot fig 2 2 3
let quarterly_sales = Ndarray.reshape sales [-1; 3] (* group by quarters *)
let _ = Axes.boxplot ax3 quarterly_sales
let _ = Axes.set_title ax3 "Quarterly Comparison"

(* Summary statistics *)
let ax4 = Figure.add_subplot fig 2 2 4
let _ = Axes.text ax4 0.1 0.8 (Printf.sprintf "Mean: %.2f" mean_sales)
let _ = Axes.text ax4 0.1 0.6 (Printf.sprintf "Std: %.2f" std_sales)

(* 4. Export results *)
let _ = Figure.savefig fig "sales_analysis.png" ~dpi:300
let _ = Ndarray.Io.save_csv [|sales; months|] "processed_sales.csv"
Libraries used: ndarray for data manipulation, hugin for visualization, ndarray-io for CSV handling

Machine Learning Pipeline

Train a neural network with automatic differentiation and visualize results.

ndarray rune kaun hugin 
(* Complete ML pipeline with automatic differentiation *)
open Rune
open Kaun
open Hugin

(* 1. Load and prepare data *)
let (x_train, y_train), (x_test, y_test) = 
  Datasets.load_mnist () |> Datasets.normalize

(* 2. Define model architecture *)
let model = Sequential.[
  Dense.create ~units:128 ~activation:Relu;
  Dropout.create ~rate:0.2;
  Dense.create ~units:64 ~activation:Relu;
  Dense.create ~units:10 ~activation:Softmax;
]

(* 3. Define loss and optimizer *)
let loss_fn = Loss.categorical_crossentropy
let optimizer = Optimizer.adam ~learning_rate:0.001 ()

(* 4. Training loop with automatic differentiation *)
let train_step model x y =
  let loss_and_grads params =
    let predictions = Model.forward model params x in
    let loss = loss_fn predictions y in
    loss
  in
  let grads = grad loss_and_grads (Model.parameters model) in
  Optimizer.apply optimizer grads

(* 5. Train and track metrics *)
let train_losses = ref []
let val_accuracies = ref []

for epoch = 1 to 10 do
  (* Training *)
  let epoch_loss = train_step model x_train y_train in
  
  (* Validation *)
  let predictions = Model.forward model (Model.parameters model) x_test in
  let accuracy = Metrics.accuracy predictions y_test in
  
  train_losses := epoch_loss :: !train_losses;
  val_accuracies := accuracy :: !val_accuracies;
  
  Printf.printf "Epoch %d: Loss %.4f, Val Acc %.4f\n" epoch epoch_loss accuracy
done

(* 6. Visualize training progress *)
let fig = Figure.create ~figsize:(12.0, 5.0) ()

(* Loss curve *)
let ax1 = Figure.add_subplot fig 1 2 1
let epochs = Ndarray.arange 1 11 1 int32
let _ = Axes.plot ax1 epochs (List.rev !train_losses |> Ndarray.of_list) ~label:"Training Loss"
let _ = Axes.set_title ax1 "Training Loss"
let _ = Axes.set_xlabel ax1 "Epoch"

(* Accuracy curve *)
let ax2 = Figure.add_subplot fig 1 2 2
let _ = Axes.plot ax2 epochs (List.rev !val_accuracies |> Ndarray.of_list) ~label:"Validation Accuracy" ~color:"orange"
let _ = Axes.set_title ax2 "Validation Accuracy"
let _ = Axes.set_xlabel ax2 "Epoch"

let _ = Figure.savefig fig "training_progress.png"
Libraries used: rune for auto-diff, kaun for neural networks, hugin for training visualization

Computer Vision Pipeline

Process images, detect objects, and train computer vision models with GPU acceleration.

ndarray sowilo rune hugin 
(* Complete computer vision pipeline *)
open Sowilo
open Rune
open Hugin

(* 1. Load and preprocess images *)
let images = Sowilo.load_images "dataset/" ~format:`RGB
let preprocessed = images 
  |> Sowilo.resize ~size:(224, 224)
  |> Sowilo.normalize ~mean:[0.485; 0.456; 0.406] ~std:[0.229; 0.224; 0.225]

(* 2. Apply image processing operations *)
let edges = Sowilo.canny_edge_detection preprocessed ~low:50 ~high:150
let blurred = Sowilo.gaussian_blur preprocessed ~sigma:1.5

(* 3. Object detection with trained model *)
let detector = Sowilo.load_yolo_model "models/yolo_v5.onnx"
let detections = Sowilo.detect_objects detector preprocessed ~confidence:0.5

(* 4. Training a custom classifier *)
let model = Sowilo.Models.resnet18 ~num_classes:10 ()
let train_data, val_data = Sowilo.load_dataset "classification_data/" ~split:0.8

let loss_fn = Sowilo.Losses.cross_entropy
let optimizer = Sowilo.Optimizers.adam ~lr:0.001 ()

(* Training with automatic differentiation *)
let train_step model batch =
  let loss_and_acc params =
    let predictions = Sowilo.Models.forward model params batch.images in
    let loss = loss_fn predictions batch.labels in
    let acc = Sowilo.accuracy predictions batch.labels in
    (loss, acc)
  in
  let (loss, acc), grads = value_and_grad loss_and_acc (Sowilo.Models.parameters model) in
  Sowilo.Optimizers.apply optimizer grads;
  (loss, acc)

(* 5. Visualization and analysis *)
let fig = Figure.create ~figsize:(15.0, 10.0) ()

(* Show original and processed images *)
let ax1 = Figure.add_subplot fig 2 3 1
let _ = Axes.imshow ax1 (Sowilo.to_ndarray (List.hd images))
let _ = Axes.set_title ax1 "Original"

let ax2 = Figure.add_subplot fig 2 3 2
let _ = Axes.imshow ax2 (Sowilo.to_ndarray edges) ~cmap:"gray"
let _ = Axes.set_title ax2 "Edge Detection"

(* Show detection results *)
let ax3 = Figure.add_subplot fig 2 3 3
let _ = Sowilo.draw_bounding_boxes ax3 (List.hd images) detections
let _ = Axes.set_title ax3 "Object Detection"

let _ = Figure.savefig fig "cv_pipeline_results.png"
Libraries used: sowilo for computer vision, rune for auto-diff, hugin for visualization

# Exploratory Data Analysis

Let's explore our dataset using the Raven ecosystem.

```ocaml
open Ndarray
open Hugin

(* Load and inspect data *)
let data = Ndarray.Io.load_csv "data.csv" ~has_header:true
let () = Ndarray.print_info data
```

## Quick Statistics

```ocaml
let column_a = Ndarray.get_column data 0
let mean_a = Ndarray.mean column_a
let std_a = Ndarray.std column_a

Printf.printf "Column A: mean=%.2f, std=%.2f\n" mean_a std_a
```

## Visualization

```ocaml
let fig = Figure.create ()
let ax = Figure.add_subplot fig 1 1 1

let _ = Axes.hist ax column_a ~bins:20 ~alpha:0.7
let _ = Axes.axvline ax mean_a ~color:"red" ~label:"Mean"
let _ = Axes.set_title ax "Distribution of Column A"
let _ = Axes.legend ax ()

Figure.show fig (* Display inline in notebook *)
```

The distribution shows a clear normal pattern with mean `mean_a`.
