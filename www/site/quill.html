data_analysis.md

Data Analysis with Raven

This notebook demonstrates loading data, performing analysis, and creating visualizations.

OCaml
open Ndarray
open Hugin
open Ndarray_datasets

(* Load the Iris dataset *)
let data = Iris.load ()
let features = data.features
let labels = data.labels
Output
Dataset loaded: 150 samples, 4 features
Shape: [150; 4]
Classes: setosa, versicolor, virginica
OCaml
(* Create a scatter plot *)
let fig = Figure.create ()
let ax = Figure.add_subplot fig 1 1 1

let x = Ndarray.slice features [All; Index 0]
let y = Ndarray.slice features [All; Index 1]

let _ = Axes.scatter ax x y ~c:labels ~alpha:0.7
let _ = Axes.set_title ax "Iris Dataset"
let _ = Figure.show fig
