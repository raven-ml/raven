Quick Example

(* Create a simple line plot *)
open Hugin
open Ndarray

(* Generate data *)
let x = Ndarray.linspace 0.0 10.0 100 float64
let y = Ndarray.sin x

(* Create figure and plot *)
let fig = Figure.create ()
let ax = Figure.add_subplot fig 1 1 1
let _ = Axes.plot ax x y ~label:"sin(x)"

(* Customize and save *)
let _ = Axes.set_title ax "Sine Wave"
let _ = Axes.set_xlabel ax "x"
let _ = Axes.set_ylabel ax "y"
let _ = Axes.legend ax ()
let _ = Figure.savefig fig "sine_wave.png"

Installation

Via OPAM

$ opam install hugin

Includes Cairo and SDL dependencies automatically.

System Dependencies

# macOS
$ brew install cairo sdl2
# Ubuntu
$ apt install libcairo2-dev libsdl2-dev

Required for Cairo and SDL backends.
