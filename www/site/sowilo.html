Differentiable Image Processing

(* Differentiable computer vision pipeline *)
open Rune
open Sowilo

(* Define a learnable image processing pipeline *)
let enhance_image params img =
  img
  |> gaussian_blur ~sigma:params.blur_sigma
  |> contrast_adjust ~factor:params.contrast
  |> edge_enhance ~strength:params.edge_strength
  |> color_balance ~params:params.color_params

(* Loss function for image enhancement *)
let enhancement_loss params (input, target) =
  let enhanced = enhance_image params input in
  l2_loss enhanced target

(* Optimize the pipeline parameters *)
let train_step params batch =
  let grad_fn = grad enhancement_loss in
  let grads = grad_fn params batch in
  Adam.update params grads ~lr:0.001

(* JIT compile for GPU acceleration *)
let fast_enhance = jit enhance_image
let fast_train = jit train_step

Via OPAM

# macOS Metal
$ opam install sowilo-metal

$ opam install sowilo-cuda
