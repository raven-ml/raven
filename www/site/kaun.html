High-level neural network library built on Rune. The Flax equivalent for OCaml with modular architecture and training utilities.

Build Your First Neural Network

(* Define a convolutional neural network *)
open Rune
open Kaun

(* Model definition using Sequential API *)
let model = Sequential.[
  Conv2d.create ~in_channels:3 ~out_channels:32 ~kernel_size:3;
  Activation.relu;
  MaxPool2d.create ~kernel_size:2;
  
  Conv2d.create ~in_channels:32 ~out_channels:64 ~kernel_size:3;
  Activation.relu;
  MaxPool2d.create ~kernel_size:2;
  
  Flatten.create ();
  Linear.create ~in_features:1600 ~out_features:128;
  Activation.relu;
  Dropout.create ~p:0.5;
  Linear.create ~in_features:128 ~out_features:10;
]

(* Training configuration *)
let trainer = Trainer.create
  ~model
  ~optimizer:(Adam.create ~lr:0.001)
  ~loss:CrossEntropy.create
  ~metrics:[Accuracy.create ()]

(* Train the model *)
let trained_model = Trainer.fit trainer
  ~train_data:cifar10_train
  ~val_data:cifar10_val
  ~epochs:50

Advanced Training Features

Learning Rate Schedules

Cosine annealing, step decay, exponential decay, and custom schedules.

Schedule.cosine_annealing ~max_lr:0.1 ~min_lr:1e-6

Model Checkpointing

Automatic saving and loading of model weights and training state.

Checkpoint.save trainer "model_epoch_10.ckpt"

Early Stopping

Prevent overfitting with patience-based early stopping and validation monitoring.

EarlyStopping.create ~patience:10 ~monitor:"val_loss"

Mixed Precision

Automatic mixed precision training for faster training and reduced memory usage.

Trainer.create ~mixed_precision:true

Gradient Clipping

Gradient norm and value clipping to stabilize training of deep networks.

~grad_clip:(ClipNorm 1.0)

Transfer Learning

Load pre-trained weights and fine-tune for new tasks with frozen layers.

Model.load_pretrained "resnet50" |> freeze_backbone
